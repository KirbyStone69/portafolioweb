% --- Slide 3: Problema ---
\begin{frame}{Problema Identificado}
    \begin{itemize}
        \item \textbf{Concentración de trabajo} en pocos integrantes en proyectos universitarios de programación
        
        \vspace{0.3cm}
        
        \item \textbf{Limitaciones del aprendizaje} colaborativo por distribución desigual de tareas
        
        \vspace{0.3cm}
        
        \item \textbf{Falta de capacitación} estructurada en herramientas de control de versiones
        
        \vspace{0.3cm}
        
        \item \textbf{Evaluación poco objetiva} del desempeño individual en equipos
    \end{itemize}
\end{frame}

% --- Slide 4: Objetivos ---
\begin{frame}{Objetivos de Investigación}
    \textbf{Objetivo General:}
    \begin{itemize}
        \item Evaluar el impacto de un sistema de control de versiones con entorno de aprendizaje interactivo en la distribución del trabajo universitario
    \end{itemize}
    
    \vspace{0.4cm}
    
    \textbf{Objetivos Específicos:}
    \begin{itemize}
        \item Desarrollar bootcamp interactivo para estandarizar competencias en Git
        \item Integrar control de versiones como herramienta de gestión académica
        \item Analizar participación grupal mediante métricas objetivas
    \end{itemize}
\end{frame}

% --- Slide 5: Marco Teórico ---
\begin{frame}{Marco Teórico}
    \textbf{Bases conceptuales:}
    \begin{itemize}
        \item \textbf{Sistemas de Control de Versiones:} Git como herramienta de colaboración
        \item \textbf{Aprendizaje Colaborativo:} Teorías de trabajo en equipo educativo
        \item \textbf{Bootcamps de Programación:} Metodologías de enseñanza intensiva
    \end{itemize}
\end{frame}

% --- Slide 6: Metodología ---
\begin{frame}{Metodología}
    \textbf{Diseño de Investigación:}
    \begin{itemize}
        \item \textbf{Tipo:} Cuasi-experimental con grupos intactos
        \item \textbf{Enfoque:} Mixto (cuantitativo + cualitativo)
        \item \textbf{Población:} Estudiantes de informática en proyectos de equipo
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{Variables:}
    \begin{itemize}
        \item \textbf{Independiente:} Bootcamp "Malleus Codeficarum" + uso obligatorio de Git
        \item \textbf{Dependiente:} Distribución equitativa del trabajo
    \end{itemize}
\end{frame}

% --- Slide 7: Instrumentos ---
\begin{frame}{Instrumentos de Recolección}
    \begin{enumerate}
        \item \textbf{Minería de Datos (Git):}
        \begin{itemize}
            \item Commits, líneas de código, frecuencia de actividad
            \item Coeficiente de Gini para medir desigualdad
        \end{itemize}
        
        \vspace{0.2cm}
        
        \item \textbf{Encuestas Likert:}
        \begin{itemize}
            \item Pre-test y post-test de competencias técnicas
        \end{itemize}
        
        \vspace{0.2cm}
        
        \item \textbf{Gráficos de Red:}
        \begin{itemize}
            \item Análisis visual de interacciones en repositorios
        \end{itemize}
    \end{enumerate}
\end{frame}

% --- Slide 8: Fases del Proyecto ---
\begin{frame}{Fases del Proyecto}
    \begin{enumerate}
        \item \textbf{Diagnóstico:} Pre-test de conocimientos previos en Git
        
        \vspace{0.3cm}
        
        \item \textbf{Intervención:} Capacitación mediante bootcamp interactivo
        
        \vspace{0.3cm}
        
        \item \textbf{Desarrollo:} Ejecución de proyectos con monitoreo continuo
        
        \vspace{0.3cm}
        
        \item \textbf{Evaluación:} Comparación estadística entre grupos
    \end{enumerate}

% --- Slide 9: Plataforma Desarrollada ---
\begin{frame}{Plataforma "Malleus Codeficarum"}
    \textbf{Estructura del Bootcamp:}
    \begin{itemize}
        \item \textbf{Área Teórica:} Explicaciones paso a paso de conceptos Git
        \item \textbf{Área Práctica:} Terminal interactiva para ejecutar comandos
        \item \textbf{Evaluación:} Seguimiento automático del progreso
    \end{itemize}
    
    \vspace{0.2cm}
    
    \textbf{Módulos incluidos:}
    \begin{itemize}
        \item Instalación y configuración de Git
        \item Repositorios: init, clone, staging
        \item Ramas, merge y resolución de conflictos
        \item Trabajo remoto: push, pull, colaboración
    \end{itemize}

% --- Slide 10: Resultados Esperados ---
\begin{frame}{Resultados Esperados}
    \textbf{Hipótesis:} La implementación de Git + capacitación reduce la concentración de trabajo
    
    \vspace{0.4cm}
    
    \textbf{Métricas esperadas:}
    \begin{itemize}
        \item Reducción del Coeficiente de Gini en contribuciones
        \item Mayor participación activa de todos los integrantes
        \item Mejora en competencias técnicas (pre/post test)
        \item Transparencia en evaluación individual
    \end{itemize}
\end{frame>

% --- Slide 11: Impacto y Viabilidad ---
\begin{frame}{Impacto y Viabilidad}
    \textbf{Impacto Académico:}
    \begin{itemize}
        \item Mejora del aprendizaje colaborativo
        \item Evaluación objetiva del desempeño individual
        \item Preparación para entornos profesionales
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{Viabilidad:}
    \begin{itemize}
        \item Tecnologías accesibles y conocidas
        \item Integración natural a las dinámicas académicas
        \item Población adecuada para la investigación
        \item Desarrollo gradual y escalable
    \end{itemize}
\end{frame}

% --- Slide 12: Conclusiones ---
\begin{frame}{Conclusiones}
    \begin{itemize}
        \item \textbf{Necesidad real:} Problema identificado en proyectos universitarios
        
        \vspace{0.3cm}
        
        \item \textbf{Solución viable:} Combinación de tecnología educativa y herramientas profesionales
        
        \vspace{0.3cm}
        
        \item \textbf{Metodología sólida:} Diseño cuasi-experimental con instrumentos validados
        
        \vspace{0.3cm}
        
        \item \textbf{Impacto esperado:} Mejora del aprendizaje colaborativo y preparación profesional
    \end{itemize}
\end{frame}
